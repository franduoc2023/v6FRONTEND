import { Component, OnInit, inject } from '@angular/core';
import { CommonModule, NgIf, NgForOf } from '@angular/common';
import {
  IonHeader,
  IonToolbar,
  IonTitle,
  IonContent,
  IonButton,
  IonThumbnail
} from '@ionic/angular/standalone';

import {
  UserService,
  WishlistItemDto
} from '../services/user.service';

import {
  CatalogService,
  WineDto,
  CheeseDto
} from '../services/catalog.service';

import { forkJoin } from 'rxjs';

export interface WishlistViewItem {
  wishlistId: string;
  productId: string;
  productType: 'WINE' | 'CHEESE' | null;
  createdAt?: string;

  name: string;
  description?: string | null;
  imageUrl?: string | null;
  price?: number | null;
  origin?: string | null;

    removing?: boolean;

}

@Component({

export class WishlistPage implements OnInit {

  private userService = inject(UserService);
  private catalogService = inject(CatalogService);

  items: WishlistViewItem[] = [];
  loading = false;

  ngOnInit(): void {
    this.loadWishlist();
  }

  loadWishlist() {
    this.loading = true;

    this.userService.getMyWishlist().subscribe({
      next: (wishlist: WishlistItemDto[]) => {
        console.log('[WishlistPage] Wishlist raw:', wishlist);

        if (!wishlist || wishlist.length === 0) {
          this.items = [];
          this.loading = false;
          console.log('[WishlistPage] Wishlist vac√≠a');
          return;
        }

        forkJoin({
          wines: this.catalogService.getWines(),
          cheeses: this.catalogService.getCheeses()
        }).subscribe({
          next: ({ wines, cheeses }: { wines: WineDto[]; cheeses: CheeseDto[] }) => {
            console.log('[WishlistPage] wines:', wines.length, 'cheeses:', cheeses.length);

            this.items = wishlist.map((w: WishlistItemDto) => {
              let name = w.productId;
              let description: string | null = null;
              let imageUrl: string | null = null;
              let price: number | null = null;
              let origin: string | null = null;
              let resolvedType: 'WINE' | 'CHEESE' | null = null;

              // 1) buscar como vino
              const wine = wines.find((x: WineDto) => x.id === w.productId);
              if (wine) {
                resolvedType = 'WINE';
                name = wine.nameEn;
                description = wine.descriptionEn ?? null;
                imageUrl = wine.imageUrl ?? null;
                price = wine.price ?? null;
                origin = wine.origin ?? null;
              } else {
                // 2) probar como queso
                const cheese = cheeses.find((x: CheeseDto) => x.id === w.productId);
                if (cheese) {
                  resolvedType = 'CHEESE';
                  name = cheese.nameEn;
                  description = cheese.descriptionEn ?? null;
                  imageUrl = cheese.imageUrl ?? null;
                  price = cheese.price ?? null;
                  origin = cheese.origin ?? null;
                }
              }

              const viewItem: WishlistViewItem = {
                wishlistId: w.id,
                productId: w.productId,
                productType: resolvedType,
                createdAt: w.createdAt,
                name,
                description,
                imageUrl,
                price,
                origin
              };

              return viewItem;
            });

            console.log('[WishlistPage] Wishlist view items:', this.items);
            this.loading = false;
          },
          error: (err: unknown) => {
            console.error('[WishlistPage] Error cargando cat√°logo para wishlist', err);

            this.items = wishlist.map((w: WishlistItemDto) => ({
              wishlistId: w.id,
              productId: w.productId,
              productType: null,
              createdAt: w.createdAt,
              name: w.productId,
              description: null,
              imageUrl: null,
              price: null,
              origin: null
            }));

            this.loading = false;
          }
        });
      },
      error: (err: unknown) => {
        console.error('[WishlistPage] Error cargando wishlist', err);
        this.loading = false;
      }
    });
  }

removingIds: Set<string> = new Set<string>();

remove(item: WishlistViewItem & { removing?: boolean }) {

  item.removing = true;

  setTimeout(() => {
    const backup = [...this.items];
    this.items = this.items.filter(i => i.wishlistId !== item.wishlistId);

    this.userService.removeFromWishlist(item.wishlistId).subscribe({
      next: () => {},
      error: err => {
        console.error('[WishlistPage] Error removing wishlist item', err);
        this.items = backup;
      }
    });
  }, 350);
}

}
